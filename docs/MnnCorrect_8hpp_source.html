280 // Function name decays to a function pointer, should be callable by just doing builder().
281 auto builder = (approximate ? approximate_builder : exact_builder);
282
283 if (order == NULL) {
284 if (automatic_order) {

383 template<typename Batch>
384 Details run(int ndim, size_t nobs, const Float* input, const Batch* batch, Float* output, const int* order = NULL) {
385 const Batch nbatches = (nobs ? *std::max_element(batch, batch + nobs) + 1 : 0);
386 std::vector<size_t> sizes(nbatches);
387 for (size_t o = 0; o < nobs; ++o) {
388 ++sizes[batch[o]];
389 }
390
391 // Avoiding the need to allocate a temporary buffer
392 // if we're already dealing with contiguous batches.
393 bool already_sorted = true;
394 for (size_t o = 1; o < nobs; ++o) {
395 if (batch[o] < batch[o-1]) {
396 already_sorted = false;
397 break;
398 }
399 }
400 if (already_sorted) {
401 return run(ndim, sizes, input, output, order);
402 }
403
404 size_t accumulated = 0;
405 std::vector<size_t> offsets(nbatches);
406 for (size_t b = 0; b < nbatches; ++b) {
407 offsets[b] = accumulated;
408 accumulated += sizes[b];
409 }
410
411 // Dumping everything by order into another vector.
412 std::vector<Float> tmp(ndim * nobs);
413 std::vector<const Float*> ptrs(nbatches, tmp.data());
414 for (size_t b = 0; b < nbatches; ++b) {
415 ptrs[b] += offsets[b] * ndim;
416 }
417 for (size_t o = 0; o < nobs; ++o) {
418 auto current = input + o * ndim;
419 auto& offset = offsets[batch[o]];
420 auto destination = tmp.data() + ndim * offset;
421 std::copy(current, current + ndim, destination);
422 ++offset;
423 }
424
425 auto stats = run_internal(ndim, sizes, ptrs, output, order);
426 restore_order(ndim, stats.merge_order, sizes, batch, output);
427 return stats;
428 }
