13/*
14 * This computes a robust average after trimming away observations that are
15 * furthest from the mean observation. Specifically, we keep all observations
16 * that are less than or equal to the '1 - trim' quantile, given a trimming
17 * proportion between 0 and 1. The trimmed set is used to compute a new mean
18 * and this process is repeated for the specified number of iterations.
19 *
20 * The use of the quantile here is approximately equivalent to removing 'trim'
21 * furthest observations, but with the following subtleties:
22 *
23 * - If there are observations at tied distances at the trim boundary,
24 * all ties are retained. This avoids arbitrary changes in the resullts
25 * depending on the input order of observations.
26 * - If trim = 1, the point closest to the mean is always retained,
27 * ensuring that the mean calculation in the next iteration is defined.
28 * - If trim > 0, the furthest point is always removed. This ensures that
29 * some trimming is always performed (unless, of course, there was only
30 * one point, in which case that point is just retained).
31 */
32template<typename Index, typename Float>
33class RobustAverage {
34public:
35 RobustAverage(int it, double tr) : iterations(it), trim(tr) {
36 if (trim < 0 || trim > 1) {
37 throw std::runtime_error("trimming proportion must be in [0, 1]");
38 }
39 if (iterations < 0) {
40 throw std::runtime_error("number of iterations must be non-negative");
41 }
42 }
43
44private:
45 int iterations;
46 double trim;
47
48 template<class Function>
49 void run(int ndim, size_t npts, Function indfun, const Float* data, Float* output) {
50 std::fill(output, output + ndim, 0);
51 for (size_t i = 0; i < npts; ++i) {
52 auto dptr = data + indfun(i) * ndim;
53 for (int d = 0; d < ndim; ++d) {
54 output[d] += dptr[d];
55 }
56 }
57 for (int d = 0; d < ndim; ++d) {
58 output[d] /= npts;
59 }
60
61 // The 'npts - 1' reflects the fact that we're comparing to a quantile.
62 // The closest point is at 0%, while the furthest point is at 100%,
63 // so we already spent one observation defining the boundaries.
64 const double threshold = static_cast<double>(npts - 1) * (1 - trim);
65
66 // And now iterating.
67 deltas.reserve(npts);
68
69 for (int it = 0; it < iterations; ++it) {
70 deltas.clear();
71 for (size_t i = 0; i < npts; ++i) {
72 auto j = indfun(i);
73 auto dptr = data + j * ndim;
74
75 Float d2 = 0;
76 for (int d = 0; d < ndim; ++d) {
77 Float diff = output[d] - dptr[d];
78 d2 += diff * diff;
79 }
80 deltas.emplace_back(d2, j);
81 }
82
83 std::sort(deltas.begin(), deltas.end());
84
85 // We always keep at least the closest observation.
86 auto first_ptr = data + deltas.front().second * ndim;
87 std::copy(first_ptr, first_ptr + ndim, output);
88 double counter = 1;
89 Float last = deltas.front().first;
90
91 // Checking if we can add another observation without cutting into
92 // the specified trim proportion - 'counter/(npt - 1)' is the
93 // quantile of the current observation in the loop. The exception
94 // is if the threshold interrupts some ties, in which case all of
95 // them are retained to avoid arbitrary ordering effects.
96 for (size_t x = 1; x < npts; ++x) {
97
98 // When considering ties, we need to account for numerical
99 // imprecision in the distance calculations. We do so by
100 // allowing a tolerance in the comparison - in this case, of
101 // 1e-10. To avoid a sliding slope of inclusion, we fix our
102 // comparisons to the first element of a tied run and only
103 // consider subsequent elements to be tied if they are within
104 // the tolerance of the first element.
105 constexpr Float tol = 1.0000000001;
106 if (deltas[x].first > last * tol) { // i.e., not tied.
107 last = deltas[x].first;
108 if (counter > threshold) {
109 break;
110 }
111 }
112
113 auto dptr = data + deltas[x].second * ndim;
114 for (int d = 0; d < ndim; ++d) {
115 output[d] += dptr[d];
116 }
117
118 ++counter;
119 }
120
121 for (int d = 0; d < ndim; ++d) {
122 output[d] /= counter;
123 }
124 }
125 }
126
127public:
128 void run(int ndim, size_t npts, const Float* data, Float* output) {
129 run(ndim, npts, [](size_t i) -> size_t { return i; }, data, output);
130 return;
131 }

133 void run(int ndim, const std::vector<Index>& indices, const Float* data, Float* output) {
134 run(ndim, indices.size(), [&](size_t i) -> size_t { return indices[i]; }, data, output);
135 return;
136 }
137
138private:
139 std::vector<std::pair<Float, Index> > deltas;
140};
