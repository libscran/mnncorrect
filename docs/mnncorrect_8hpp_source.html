206void compute(const std::size_t num_dim, const Index_ num_obs, const Float_* const input, const Batch_* const batch, Float_* const output, const Options<Index_, Float_, Matrix_>& options) {
207 const BatchIndex nbatches = (num_obs ? sanisizer::sum<BatchIndex>(*std::max_element(batch, batch + num_obs), 1) : static_cast<BatchIndex>(0));
208 auto sizes = sanisizer::create<std::vector<Index_> >(nbatches);
209 for (Index_ o = 0; o < num_obs; ++o) {
210 ++sizes[batch[o]];
211 }
212
213 // Avoiding the need to allocate a temporary buffer
214 // if we're already dealing with contiguous batches.
215 bool already_sorted = true;
216 for (Index_ o = 1; o < num_obs; ++o) {
217 if (batch[o] < batch[o-1]) {
218 already_sorted = false;
219 break;
220 }
221 }
222 if (already_sorted) {
223 compute(num_dim, sizes, input, output, options);
224 return;
225 }
226
227 Index_ accumulated = 0;
228 auto offsets = sanisizer::create<std::vector<Index_> >(nbatches);
229 std::vector<Float_> tmp(sanisizer::product<typename std::vector<Float_>::size_type>(num_dim, num_obs));
230 auto ptrs = sanisizer::create<std::vector<const Float_*> >(nbatches);
231 for (BatchIndex b = 0; b < nbatches; ++b) {
232 ptrs[b] = tmp.data() + sanisizer::product_unsafe<std::size_t>(accumulated, num_dim);
233 offsets[b] = accumulated;
234 accumulated += sizes[b]; // this won't overflow as know that num_obs fits in an Index_.
235 }
236
237 for (Index_ o = 0; o < num_obs; ++o) {
238 auto& offset = offsets[batch[o]];
239 std::copy_n(
240 input + sanisizer::product_unsafe<std::size_t>(o, num_dim),
241 num_dim,
242 tmp.data() + sanisizer::product_unsafe<std::size_t>(offset, num_dim)
243 );
244 ++offset;
245 }
246
247 internal::compute(num_dim, sizes, ptrs, output, options);
248 internal::restore_input_order(num_dim, sizes, batch, output);
249}
