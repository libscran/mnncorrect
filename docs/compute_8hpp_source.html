193template<typename Index_, typename Float_, typename Batch_, class Matrix_>
194Details compute(size_t num_dim, size_t num_obs, const Float_* input, const Batch_* batch, Float_* output, const Options<Index_, Float_, Matrix_>& options) {
195 const size_t nbatches = (num_obs ? static_cast<size_t>(*std::max_element(batch, batch + num_obs)) + 1 : 0);
196 std::vector<size_t> sizes(nbatches);
197 for (size_t o = 0; o < num_obs; ++o) {
198 ++sizes[batch[o]];
199 }
200
201 // Avoiding the need to allocate a temporary buffer
202 // if we're already dealing with contiguous batches.
203 bool already_sorted = true;
204 for (size_t o = 1; o < num_obs; ++o) {
205 if (batch[o] < batch[o-1]) {
206 already_sorted = false;
207 break;
208 }
209 }
210 if (already_sorted) {
211 return compute(num_dim, sizes, input, output, options);
212 }
213
214 size_t accumulated = 0;
215 std::vector<size_t> offsets(nbatches);
216 for (size_t b = 0; b < nbatches; ++b) {
217 offsets[b] = accumulated;
218 accumulated += sizes[b];
219 }
220
221 // Dumping everything by order into another vector.
222 std::vector<Float_> tmp(num_dim * num_obs);
223 std::vector<const Float_*> ptrs(nbatches);
224 for (size_t b = 0; b < nbatches; ++b) {
225 ptrs[b] = tmp.data() + offsets[b] * num_dim;
226 }
227
228 for (size_t o = 0; o < num_obs; ++o) {
229 auto current = input + o * num_dim;
230 auto& offset = offsets[batch[o]];
231 auto destination = tmp.data() + num_dim * offset; // already size_t's, so no need to cast to avoid overflow.
232 std::copy_n(current, num_dim, destination);
233 ++offset;
234 }
235
236 auto stats = internal::compute(num_dim, sizes, ptrs, output, options);
237 internal::restore_order(num_dim, stats.merge_order, sizes, batch, output);
238 return stats;
239}
