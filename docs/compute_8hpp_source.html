198template<typename Index_, typename Float_, typename Batch_, class Matrix_>
199Details compute(std::size_t num_dim, Index_ num_obs, const Float_* input, const Batch_* batch, Float_* output, const Options<Index_, Float_, Matrix_>& options) {
200 const internal::BatchIndex nbatches = (num_obs ? static_cast<internal::BatchIndex>(*std::max_element(batch, batch + num_obs)) + 1 : 0);
201 std::vector<Index_> sizes(nbatches);
202 for (Index_ o = 0; o < num_obs; ++o) {
203 ++sizes[batch[o]];
204 }
205
206 // Avoiding the need to allocate a temporary buffer
207 // if we're already dealing with contiguous batches.
208 bool already_sorted = true;
209 for (Index_ o = 1; o < num_obs; ++o) {
210 if (batch[o] < batch[o-1]) {
211 already_sorted = false;
212 break;
213 }
214 }
215 if (already_sorted) {
216 return compute(num_dim, sizes, input, output, options);
217 }
218
219 std::size_t accumulated = 0; // use size_t to avoid overflow issues during later multiplication.
220 std::vector<std::size_t> offsets(nbatches);
221 for (internal::BatchIndex b = 0; b < nbatches; ++b) {
222 offsets[b] = accumulated;
223 accumulated += sizes[b];
224 }
225
226 // Dumping everything by order into another vector.
227 std::vector<Float_> tmp(num_dim * static_cast<std::size_t>(num_obs)); // cast to size_t to avoid overflow.
228 std::vector<const Float_*> ptrs(nbatches);
229 for (internal::BatchIndex b = 0; b < nbatches; ++b) {
230 ptrs[b] = tmp.data() + offsets[b] * num_dim; // already size_t's, so no need to cast to avoid overflow.
231 }
232
233 for (Index_ o = 0; o < num_obs; ++o) {
234 auto current = input + static_cast<std::size_t>(o) * num_dim; // cast to size_t to avoid overflow.
235 auto& offset = offsets[batch[o]];
236 auto destination = tmp.data() + num_dim * offset; // already size_t's, so no need to cast to avoid overflow.
237 std::copy_n(current, num_dim, destination);
238 ++offset;
239 }
240
241 auto stats = internal::compute(num_dim, sizes, ptrs, output, options);
242 internal::restore_order(num_dim, stats.merge_order, sizes, batch, output);
243 return stats;
244}
